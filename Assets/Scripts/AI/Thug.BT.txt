tree("Root")
	parallel
		repeat mute tree("Live")
		repeat mute tree("DIE")

tree("Live")
	fallback
		while LowOnAmmo
			tree("FindCoverAndReload")
		tree("FireOnSight")
		tree("MoveIntoAttackRangeOfPlayer")
		SecurePinging

tree("FireOnSight")
	repeat
		sequence
			PlayerInSight
			PlayerInAttackRange
			not LowOnAmmo
			Shoot

tree("DIE")
	sequence
		HasZeroHealth
		DestroySelf

tree("FindCoverAndReload")
	sequence
		random
			Reload
			sequence
				SetMovementTargetToCover
				tree("MoveToMovementTarget")
				Reload

tree("MoveToMovementTarget")
	sequence
		RecalculatePathToMovementTarget
		fallback
			repeat
				MoveAlongPath
			Succeed

tree("MoveIntoAttackRangeOfPlayer")
	parallel
		repeat
			sequence
				PlayerInSight
				SetMovementTargetToPlayer
				RecalculatePathToMovementTarget
				mute 
					repeat
						sequence
							MoveAlongPath
							not PlayerTooFarFromEndOfPath
		fallback
			repeat
				fallback
					not PlayerInSight
					not PlayerInAttackRange
			Succeed
