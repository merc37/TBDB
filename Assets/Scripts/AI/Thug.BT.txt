tree("Root")
	parallel
		repeat mute tree("Live")
		repeat mute tree("DIE")

tree("Live")
	parallel
		repeat
			mute
				tree("FireOnSight")
		repeat
			mute
				sequence
						ShouldRoll
						not IsRollOnCooldown
						StopMovement
						Roll
		repeat
			mute
				fallback
					sequence
						tree("PlayerInSight")
						PlayerInMedAttackRange
						not PlayerRunningAway
						StopMovement
					sequence
						tree("PlayerInSight")
						SetMovementTargetToPlayerPosition
						SetLookTargetToMovementDirection
						PathToMovementTarget
					sequence
						SetMovementTargetToPlayerLastKnownPosition
						SetLookTargetToMovementDirection
						fallback
							PathToMovementTarget
							sequence
								Stop
								UnSetPlayerLastKnownPosition
								SetLookTargetToPlayerLastKnownHeading
								UnSetPlayerLastKnownHeading
							
tree("FireOnSight")
	sequence
		tree("PlayerInSight")
		PlayerInAttackRange
		not LowOnAmmo
		AimAtPlayer
		Shoot

tree("DIE")
	sequence
		HasZeroHealth
		DestroySelf

tree("PlayerInSight")
	sequence
		PlayerInSight
		SetLookTargetToPlayer
		SetPlayerLastKnownPosition
		SetPlayerLastKnownHeading

tree("Temp")
	sequence
		sequence
			LowOnHealth
			not FoundCover
			SetMovementTargetToCover
			mute
				fallback
					repeat
						sequence
							SetLookTargetToMovementDirection
							Fail
							//MoveTowardsMovementTarget
					sequence
						Stop
						SetCoverFound
		sequence
			Fail
			LowOnAmmo
			random
				Reload
				fallback
					sequence
						SetMovementTargetToCover
						repeat
							Fail
							//MoveTowardsMovementTarget
					Reload
