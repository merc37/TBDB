//Succeeds: if trigger was pulled (no guarantee shot was actually fired, depends on gun fire rate)
//Fails: if player no longer in sight, not in attack range or low on ammo
tree("FireOnSight")
	sequence
		tree("PlayerInSight")
		not IsAmmoZero
		//not IsReloading
		PlayerInAttackRange
		AimAtPlayer
		Shoot

tree("WaitReload")
	sequence
		Reload
		repeat
			IsReloading
